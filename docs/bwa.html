5.2 比对

单端数据

DATA=data
REF=genome/Human/Homo_sapiens.GRCh38.dna.primary_assembly.fa
BWA=results/bwa
CLEANDATA=${DATA}/reads
Run=SRR1343245

make -f src/run/bwa.mk \
        REF=${REF} \
        R1=${CLEANDATA}/${Run}_1.trimmed.fastq \
        BAM=${BWA}/{sample}.bam \
        run
DATA=data
design=${DATA}/design.csv
REF=genome/Human/Homo_sapiens.GRCh38.dna.primary_assembly.fa
CLEANDATA=${DATA}/reads
BWA=results/bwa
                
cat ${design} | parallel --header : --colsep , \
        make -f src/run/bwa.mk \
        REF=${REF} \
        R1=${CLEANDATA}/{Run}_1.trimmed.fastq \
        BAM=${BWA}/{sample}.bam \
        run
